-- RELACIONAMENTOS, criados apenas se ainda não existirem
do $$
begin
  if not exists (select 1 from pg_constraint where conname = 'lessons_course_id_fkey') then
    alter table lessons
    add constraint lessons_course_id_fkey
    foreign key (course_id) references courses(id) on delete cascade;
  end if;

  if not exists (select 1 from pg_constraint where conname = 'attendance_course_id_fkey') then
    alter table attendance
    add constraint attendance_course_id_fkey
    foreign key (course_id) references courses(id) on delete cascade;
  end if;

  if not exists (select 1 from pg_constraint where conname = 'attendance_lesson_id_fkey') then
    alter table attendance
    add constraint attendance_lesson_id_fkey
    foreign key (lesson_id) references lessons(id) on delete cascade;
  end if;
end $$;
